#include <iostream>
#include <string>
#include <array>
#include <map>
#include <cstdlib>
#include <ctime>
#include <fstream>
#include <list>

using namespace std;

const int VEHICLE_TYPES = 3;

//Prototype
bool load_data(map<string, array<list<string>, VEHICLE_TYPES>> & traffic_map);

int main()
{
    map<string, array<list<string>, VEHICLE_TYPES>> traffic_map;

    cout << "Testing load_data function...\n";

    if (load_data(traffic_map))
    {
        cout << "SUCCESS: Data loaded correctly.\n";
        cout << "Number of intersections: "
             << traffic_map.size() << endl;
    }
    else
    {
        cout << "FAILURE: load_data failed.\n";
    }

    return 0;
}
